86       """Class to represent an invoice."""
87   
88       def __init__(self, order, discount=0):
89           self.__order = order
90           self.__discounts = discount  # Discounts applied
91           self.__vat_rate = 0.08  # VAT fixed rate of 8%
92   
93       # Calculate total price including VAT and discount
94       def calculate_total(self):
95           total = sum(item.get_price() for item in self.__order.get_items())
96           total -= self.__discounts
97           total += total * self.__vat_rate
98           return total
99   
100      # __str__ method to print the invoice details
101      def __str__(self):
102          return (f"Invoice:\nCustomer: {self.__order.get_customer().get_name()}\n"
103                  f"Items: {', '.join(item.get_title() for item in self.__order.get_items())}\n"
104                  f"Discounts Applied: ${self.__discounts:.2f}\n"
105                  f"Total Amount (with VAT): ${self.calculate_total():.2f}")
